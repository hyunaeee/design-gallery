즐겨찾는 유틸 함수retry.ts
export async function retry<T>(
  fn: () => Promise<T>,
  opts = { tries: 3, delayMs: 300 }
): Promise<T> {
  // 지수 백오프로 재시도하는 작은 헬퍼
  for (let i = 0; i < opts.tries; i++) {
    try { return await fn(); }
    catch (err) {
      if (i === opts.tries - 1) throw err;
      await new Promise(r => setTimeout(r, opts.delayMs * 2 ** i));
    }
  }
  throw new Error("unreachable");
}
